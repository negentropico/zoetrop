```tsx
/* Copy verbatim from RESEARCH.md lines 961–982 */
const ELEVATION = { flat: "", xs: "shadow-xs", sm: "shadow-sm", md: "shadow-md", lg: "shadow-lg" };
const PADDING = { none: "", sm: "p-3", md: "p-5", lg: "p-6" };
const TONE_BG = { energy: "bg-energy-50", vital: "bg-vital-50", focus: "bg-focus-50", mist: "bg-surface-sunken" };
const ACCENT_COLOR = { energy: "var(--energy)", vital: "var(--vital)", focus: "var(--focus)", ink: "var(--ink)" };

export function Card({ elevation = "sm", accent = null, tone = null, padding = "lg", interactive = false, onClick, children, className = "", style }: CardProps) {
  const accentColor = accent ? (ACCENT_COLOR[accent as keyof typeof ACCENT_COLOR] ?? accent) : null;
  return (
    <div
      onClick={onClick}
      className={[
        "bg-surface rounded-xl border border-border",
        ELEVATION[elevation],
        PADDING[padding],
        tone ? TONE_BG[tone] : "",
        interactive ? "cursor-pointer transition-[transform,box-shadow] hover:-translate-y-0.5 hover:shadow-md active:scale-[0.99]" : "",
        className,
      ].filter(Boolean).join(" ")}
      style={{ ...(accentColor ? { borderTop: `2px solid ${accentColor}` } : {}), ...style }}
    >
      {children}
    </div>
  );
}
```

  if (!data || data.length < 2) return null;
  const min = Math.min(...data), max = Math.max(...data), span = max - min || 1;
  const pts = data.map((v, i) => [
    (i / (data.length - 1)) * (width - 2) + 1,

      <path d={d} fill="none" stroke={color} strokeWidth="1.5" strokeLinecap="round" strokeLinejoin="round" opacity="0.7" />

### CSS-First Dark Mode Rule (all components)

**Source:** RESEARCH.md §Dark Theme + UI-SPEC §Dark Theme

```
Rule: All color must reference semantic tokens (var(--ink), var(--surface), var(--border),
var(--vital-50), etc.) — NEVER hardcode hex values in component style props or className.
The dark token remap in html[data-theme="dark"] flips values automatically.
Exception: Use "var(--n-50)" not "#f5f2f0" even when referencing neutrals directly.
No dark: utility prefix needed on any component — cascade handles it.
```
